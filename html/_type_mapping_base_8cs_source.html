185  private void AddLeftIDictionaryMapping(Type LeftType, Type RightType)
186  {
187  Contract.Requires<ArgumentNullException>(RightType != null, "RightType");
188  Contract.Requires<ArgumentNullException>(LeftType != null, "LeftType");
189  var Properties = RightType.GetProperties();
190  Parallel.For(0, Properties.Length, x =>
191  {
192  PropertyInfo Property = Properties[x];
193  var RightGet = Properties[x].PropertyGetter<Right>();
194  var RightSet = RightGet.PropertySetter<Right>().Compile();
195  var LeftProperty = LeftType.GetProperty(Property.Name);
196  if (LeftProperty != null)
197  {
198  var LeftGet = LeftProperty.PropertyGetter<Left>();
199  this.AddMapping(LeftGet, RightGet);
200  }
201  else
202  {
203  this.AddMapping(new Func<Left, object>(y =>
204  {
205  var Temp = (IDictionary<string, object>)y;
206  if (Temp.ContainsKey(Property.Name))
207  return Temp[Property.Name];
208  var Key = Temp.Keys.FirstOrDefault(z => string.Equals(z.Replace("_", ""), Property.Name, StringComparison.InvariantCultureIgnoreCase));
209  if (!string.IsNullOrEmpty(Key))
210  return Temp[Key];
211  return null;
212  }),
