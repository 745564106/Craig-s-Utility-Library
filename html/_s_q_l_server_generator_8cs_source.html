415  {

417  IProperty<T> IDProperty = ((IProperty<T>)Mapping.IDProperties.FirstOrDefault());
418  var IDValue = IDProperty.GetValue(Object).To(default(PrimaryKeyType));

431  if (Any(Param1).Execute()[0].Count == 0)

722  private static IEnumerable<IMapping> FindChildren(IMapping Mapping, Graph<IMapping> graph)
723  {
724  var ResultList = new List<IMapping>();
725  var StartingNodes = new Vertex<IMapping>[] { graph.Vertices.First(x => x.Data == Mapping) }.ToList();
726  while (StartingNodes.Count > 0)
727  {
728  var Vertex = StartingNodes.First();
729  StartingNodes.Remove(Vertex);
730  ResultList.AddIfUnique(Vertex.Data);
731  foreach (Edge<IMapping> Edge in Vertex.OutgoingEdges.ToList())
732  {
733  Vertex<IMapping> Sink = Edge.Sink;
734  Edge.Remove();
735  if (Sink.IncomingEdges.Count == 0)
736  {
737  ResultList.AddIfUnique(Sink.Data);
738  StartingNodes.AddIfUnique(Sink);
739  }
740  }
741  }
742  return ResultList;
743  }
