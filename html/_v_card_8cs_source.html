301  protected override void LoadFromData(string Data)
302  {
303  string Content = Data;
304  foreach (Match TempMatch in Regex.Matches(Content, "(?<Title>[^:]+):(?<Value>.*)"))
305  {
306  switch (TempMatch.Groups["Title"].Value.ToUpperInvariant())
307  {
308  case "N":
309  var Name = TempMatch.Groups["Value"].Value.Split(';');
310  if (Name.Length > 0)
311  {
312  LastName = Name[0];
313  if (Name.Length > 1)
314  FirstName = Name[1];
315  if (Name.Length > 2)
316  MiddleName = Name[2];
317  if (Name.Length > 3)
318  Prefix = Name[3];
319  if (Name.Length > 4)
320  Suffix = Name[4];
321  }
322 
323  break;
324  case "TEL;WORK":
325  DirectDial = TempMatch.Groups["Value"].Value;
326  break;
327  case "EMAIL;INTERNET":
328  Email = TempMatch.Groups["Value"].Value;
329  break;
330  case "TITLE":
331  Title = TempMatch.Groups["Value"].Value;
332  break;
333  case "ORG":
334  Organization = TempMatch.Groups["Value"].Value;
335  break;
336  }
337  }
