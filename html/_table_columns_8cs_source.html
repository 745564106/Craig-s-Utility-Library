60  public void GetCommand(IBatch batch)
61  {
62  if (batch == null)
63  throw new ArgumentNullException(nameof(batch));
64  batch.AddCommand(null, null, CommandType.Text, @"SELECT sys.tables.name as [Table],sys.columns.name AS [Column], sys.systypes.name AS [COLUMN_TYPE],
65  sys.columns.max_length as [MAX_LENGTH], sys.columns.is_nullable as [IS_NULLABLE],
66  sys.columns.is_identity as [IS_IDENTITY], sys.index_columns.index_id as [IS_INDEX],
67  key_constraints.name as [PRIMARY_KEY], key_constraints_1.name as [UNIQUE],
68  tables_1.name as [FOREIGN_KEY_TABLE], columns_1.name as [FOREIGN_KEY_COLUMN],
69  sys.default_constraints.definition as [DEFAULT_VALUE]
70  FROM sys.tables
71  INNER JOIN sys.columns on sys.columns.object_id=sys.tables.object_id
72  INNER JOIN sys.systypes ON sys.systypes.xtype = sys.columns.system_type_id
73  LEFT OUTER JOIN sys.index_columns on sys.index_columns.object_id=sys.tables.object_id and sys.index_columns.column_id=sys.columns.column_id
74  LEFT OUTER JOIN sys.key_constraints on sys.key_constraints.parent_object_id=sys.tables.object_id and sys.key_constraints.parent_object_id=sys.index_columns.object_id and sys.index_columns.index_id=sys.key_constraints.unique_index_id and sys.key_constraints.type='PK'
75  LEFT OUTER JOIN sys.foreign_key_columns on sys.foreign_key_columns.parent_object_id=sys.tables.object_id and sys.foreign_key_columns.parent_column_id=sys.columns.column_id
76  LEFT OUTER JOIN sys.tables as tables_1 on tables_1.object_id=sys.foreign_key_columns.referenced_object_id
77  LEFT OUTER JOIN sys.columns as columns_1 on columns_1.column_id=sys.foreign_key_columns.referenced_column_id and columns_1.object_id=tables_1.object_id
78  LEFT OUTER JOIN sys.key_constraints as key_constraints_1 on key_constraints_1.parent_object_id=sys.tables.object_id and key_constraints_1.parent_object_id=sys.index_columns.object_id and sys.index_columns.index_id=key_constraints_1.unique_index_id and key_constraints_1.type='UQ'
79  LEFT OUTER JOIN sys.default_constraints on sys.default_constraints.object_id=sys.columns.default_object_id
80  WHERE sys.systypes.xusertype <> 256");
81  }

88  private static void SetupColumns(ITable table, dynamic item)
89  {
90  if (item == null)
91  throw new ArgumentNullException(nameof(item));
92  if (table == null)
93  throw new ArgumentNullException(nameof(table));
94  if (table.ContainsColumn(item.Column))
95  {
96  table.AddForeignKey(item.Column, item.FOREIGN_KEY_TABLE, item.FOREIGN_KEY_COLUMN);
97  }
98  else
99  {
100  table.AddColumn<string>(item.Column,
101  Utilities.DataTypes.TypeConversionExtensions.To(Utilities.DataTypes.TypeConversionExtensions.To<string, SqlDbType>(item.COLUMN_TYPE), DbType.Int32),
102  (item.COLUMN_TYPE == "nvarchar") ? item.MAX_LENGTH / 2 : item.MAX_LENGTH,
103  item.IS_NULLABLE,
104  item.IS_IDENTITY,
105  item.IS_INDEX != 0,
106  !string.IsNullOrEmpty(item.PRIMARY_KEY),
107  !string.IsNullOrEmpty(item.UNIQUE),
108  item.FOREIGN_KEY_TABLE,
109  item.FOREIGN_KEY_COLUMN,
110  item.DEFAULT_VALUE);
111  }
112  }
