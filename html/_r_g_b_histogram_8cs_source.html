84  {
85  if (R[x] > 0f)

87  if (RMax < x)
88  RMax = x;
89  if (RMin > x)

91  }

93  {
94  if (GMax < x)
95  GMax = x;
96  if (GMin > x)
97  GMin = x;
98  }
99  if (B[x] > 0f)

101  if (BMax < x)

104  BMin = x;
105  }

108  float PreviousR = R[0];

110  float PreviousG = G[0];
111  G[0] = G[0] * 256 / TotalPixels;
112  float PreviousB = B[0];
113  B[0] = B[0] * 256 / TotalPixels;
114  for (int x = 1; x < 256; ++x)
115  {
116  PreviousR += R[x];
117  PreviousG += G[x];
118  PreviousB += B[x];
119  R[x] = ((PreviousR - R[RMin]) / (TotalPixels - R[RMin])) * 255;
120  G[x] = ((PreviousG - G[GMin]) / (TotalPixels - G[GMin])) * 255;
121  B[x] = ((PreviousB - B[BMin]) / (TotalPixels - B[BMin])) * 255;
122  }

131  public virtual void LoadImage(SwiftBitmap ImageUsing)
132  {
133  Contract.Requires<ArgumentNullException>(ImageUsing != null, "ImageUsing");
134  Width = ImageUsing.Width;
135  Height = ImageUsing.Height;
136  ImageUsing.Lock();
137  R.Clear();
138  G.Clear();
139  B.Clear();
140  for (int x = 0; x < ImageUsing.Width; ++x)
141  {
142  for (int y = 0; y < ImageUsing.Height; ++y)
143  {
144  var TempColor = ImageUsing.GetPixel(x, y);
145  ++R[(int)TempColor.R];
146  ++G[(int)TempColor.G];
147  ++B[(int)TempColor.B];
148  }
149  }
150  ImageUsing.Unlock();
151  }
