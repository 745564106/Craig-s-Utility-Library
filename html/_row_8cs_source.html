58  public Row(string Content, string Delimiter)
59  {
60  Contract.Requires<ArgumentNullException>(Content != null, "Content");
61  Contract.Requires<ArgumentNullException>(!string.IsNullOrEmpty(Delimiter), "Delimiter");
62  this.Cells = new List<Cell>();
63  this.Delimiter = Delimiter;
64  var TempSplitter = new Regex(string.Format(CultureInfo.InvariantCulture, "(?<Value>\"(?:[^\"]|\"\")*\"|[^{0}\r\n]*?)(?<Delimiter>{0}|\r\n|\n|$)", Regex.Escape(Delimiter)));
65  var Matches = TempSplitter.Matches(Content);
66  bool Finished = false;
67  foreach (Match Match in Matches)
68  {
69  if (!Finished)
70  {
71  Cells.Add(new Cell(Match.Groups["Value"].Value));
72  }
73  Finished = string.IsNullOrEmpty(Match.Groups["Delimiter"].Value) || Match.Groups["Delimiter"].Value == "\r\n" || Match.Groups["Delimiter"].Value == "\n";
74  }
75  }
