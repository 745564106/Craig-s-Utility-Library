149  public IList<IList<dynamic>> Execute()
150  {
151  var ReturnValue = new List<IList<dynamic>>();
152  if (Commands.Count == 0)
153  {
154  ReturnValue.Add(new List<dynamic>());
155  return ReturnValue;
156  }
157  using (DirectoryEntry Entry = new DirectoryEntry(Source.Server, Source.UserName, Source.Password, AuthenticationTypes.Secure))
158  {
159  using (DirectorySearcher Searcher = new DirectorySearcher(Entry))
160  {
161  Searcher.PageSize = 1000;
162  foreach (Command Command in Commands)
163  {
164  Searcher.Filter = Command.SQLCommand;
165  using (SearchResultCollection Results = Searcher.FindAll())
166  {
167  var ReturnValues = new List<dynamic>();
168  foreach (SearchResult Result in Results)
169  {
170  var TempValue = new Dynamo();
171  foreach (PropertyValueCollection Property in Result.GetDirectoryEntry().Properties)
172  {
173  TempValue[Property.PropertyName] = Property.Value;
174  ReturnValues.Add(TempValue);
175  }
176  }
177  ReturnValue.Add(ReturnValues);
178  }
179  }
180  }
181  Entry.Close();
182  }
183  for (int x = 0; x < Commands.Count(); ++x)
184  {
185  Commands[x].Finalize(ReturnValue[x]);
186  }
187  return ReturnValue;
188  }
