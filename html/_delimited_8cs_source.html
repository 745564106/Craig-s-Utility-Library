103  public void Parse(string FileContent)
104  {
105  if (string.IsNullOrEmpty(FileContent))
106  return;
107  var TempSplitter = new Regex("[^\"\r\n]*(\r\n|\n|$)|(([^\"\r\n]*)(\"[^\"]*\")([^\"\r\n]*))*(\r\n|\n|$)");
108  var Matches = TempSplitter.Matches(FileContent);
109  if (string.IsNullOrEmpty(Delimiter) && Matches != null)
110  Delimiter = CheckDelimiters(Matches.Where(x => !string.IsNullOrEmpty(x.Value)).FirstOrDefault().Chain(x => x.Value, ","));
111  Matches.Where(x => !string.IsNullOrEmpty(x.Value))
112  .ForEach(x => Records.Add(new Row(x.Value, Delimiter)));
113  }

170  private static string CheckDelimiters(string Content)
171  {
172  if (string.IsNullOrEmpty(Content))
173  return ",";
174  string[] Delimiters = { ",", "|", "\t", "$", ";", ":" };
175  int[] Count = new int[6];
176  int MaxIndex = 0;
177  for (int x = 0; x < Delimiters.Length; ++x)
178  {
179  string TempDelimiter = Delimiters[x];
180  var TempSplitter = new Regex(string.Format(CultureInfo.InvariantCulture, "(?<Value>\"(?:[^\"]|\"\")*\"|[^{0}\r\n]*?)(?<Delimiter>{0}|\r\n|\n|$)", Regex.Escape(TempDelimiter)));
181  Count[x] = TempSplitter.Matches(Content).Count;
182  if (Count[MaxIndex] < Count[x])
183  MaxIndex = x;
184  }
185  return Count[MaxIndex] > 1 ? Delimiters[MaxIndex] : ",";
186  }
