178  private IList<IList<dynamic>> ExecuteCommands()
179  {
180  Contract.Requires(Source != null);
181  if (Commands == null)
182  return new List<IList<dynamic>>();
183  var ReturnValue = new List<IList<dynamic>>();
184  if (Commands.Count == 0)
185  {
186  ReturnValue.Add(new List<dynamic>());
187  return ReturnValue;
188  }
189  var Factory = DbProviderFactories.GetFactory(Source.SourceType);
190  using (DbConnection Connection = Factory.CreateConnection())
191  {
192  Connection.ConnectionString = Source.Connection;
193  using (DbCommand ExecutableCommand = Factory.CreateCommand())
194  {
195  ExecutableCommand.Connection = Connection;
196  ExecutableCommand.CommandType = CommandType.Text;
197  if (Commands.Count > 1
198  && !Commands.Any(x => x.SQLCommand.Contains("ALTER DATABASE"))
199  && !Commands.Any(x => x.SQLCommand.Contains("CREATE DATABASE")))
200  ExecutableCommand.BeginTransaction();
201  ExecutableCommand.Open();
202 
203  try
204  {
205  int Count = 0;
206  while (true)
207  {
208  var FinalParameters = new List<IParameter>();
209  bool Finalizable = false;
210  string FinalSQLCommand = "";
211  int ParameterTotal = 0;
212  ExecutableCommand.Parameters.Clear();
213  for (int y = Count; y < Commands.Count; ++y)
214  {
215  ICommand Command = Commands[y];
216  if (ParameterTotal + Command.Parameters.Count > 2100)
217  break;
218  ParameterTotal += Command.Parameters.Count;
219  Finalizable |= Commands[y].Finalizable;
220  if (Command.CommandType == CommandType.Text)
221  {
222  FinalSQLCommand += string.IsNullOrEmpty(Command.SQLCommand) ?
223  "" :
224  ParameterRegex.Replace(Command.SQLCommand, x =>
225  {
226  if (!x.Value.StartsWith("@@", StringComparison.Ordinal))
227  return x.Value + "Command" + Count.ToString(CultureInfo.InvariantCulture);
228  return x.Value;
229  }) + Environment.NewLine;
230  foreach (IParameter Parameter in Command.Parameters)
231  {
232  FinalParameters.Add(Parameter.CreateCopy("Command" + Count.ToString(CultureInfo.InvariantCulture)));
233  }
234  }
235  else
236  {
237  FinalSQLCommand += Command.SQLCommand + Environment.NewLine;
238  foreach (IParameter Parameter in Command.Parameters)
239  {
240  FinalParameters.Add(Parameter.CreateCopy(""));
241  }
242  }
243  ++Count;
244  }
245 
246  ExecutableCommand.CommandText = FinalSQLCommand;
247  FinalParameters.ForEach(x => x.AddParameter(ExecutableCommand));
248 
249  using (DbDataReader TempReader = ExecutableCommand.ExecuteReader())
250  {
251  if (Finalizable)
252  {
253  ReturnValue.Add(GetValues(TempReader));
254  while (TempReader.NextResult())
255  {
256  ReturnValue.Add(GetValues(TempReader));
257  }
258  }
259  }
260  if (Count >= CommandCount)
261  break;
262  }
263  ExecutableCommand.Commit();
264  }
265  catch { ExecutableCommand.Rollback(); throw; }
266  finally { ExecutableCommand.Close(); }
267  }
268  }
269  for (int x = 0, y = 0; x < Commands.Count(); ++x)
270  {
271  if (Commands[x].Finalizable)
272  {
273  Commands[x].Finalize(ReturnValue[y]);
274  ++y;
275  }
276  else
277  Commands[x].Finalize(new List<dynamic>());
278  }
279  return ReturnValue;
280  }
