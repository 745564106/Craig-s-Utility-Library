128  public static int GreatestCommonDenominator(this int Value1, int Value2)
129  {
130  Contract.Requires<ArgumentOutOfRangeException>(Value1 != Int32.MinValue, "Value1 can not be Int32.MinValue");
131  Contract.Requires<ArgumentOutOfRangeException>(Value2 != Int32.MinValue, "Value2 can not be Int32.MinValue");
132  Value1 = Value1.Absolute();
133  Value2 = Value2.Absolute();
134  while (Value1 != 0 && Value2 != 0)
135  {
136  if (Value1 > Value2)
137  Value1 %= Value2;
138  else
139  Value2 %= Value1;
140  }
141  return Value1 == 0 ? Value2 : Value1;
142  }

231  public static T Mode<T>(this IEnumerable<T> Values)
232  {
233  if (Values == null)

238  foreach (T Value in Values)

574  public static double Variance<T>(this IEnumerable<T> Values, Func<T, decimal> Selector)
575  {
576  if (Values == null || Values.Count() == 0)
577  return 0;
578  var MeanValue = Values.Average(Selector);
579  var Sum = Values.Sum(x => (Selector(x) - MeanValue).Pow(2));
580  return Sum / (double)Values.Count();
581  }
