410  private static void CopyOrAdd(List<Dynamo> ReturnValue, IProperty IDProperty, Dynamo Item)
411  {
412  Contract.Requires<ArgumentNullException>(IDProperty != null);
413  if (Item == null)
414  return;

416  ReturnValue = new List<Dynamo>();
417  var IDValue = IDProperty.GetValue(Item);
418  var Value = ReturnValue.FirstOrDefault(x => IDProperty.GetValue(x).Equals(IDValue));
419  if (Value == null)
420  ReturnValue.Add(Item);
