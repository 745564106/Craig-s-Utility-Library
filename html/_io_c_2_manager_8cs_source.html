184  private static void LoadAssemblies(List<Assembly> Assemblies, params AssemblyName[] assemblyName)
185  {
186  if (assemblyName == null)
187  return;
188  foreach (var Name in assemblyName.Where(x => x != null
189  && !x.FullName.StartsWith("System.", StringComparison.InvariantCultureIgnoreCase)
190  && !x.FullName.StartsWith("Microsoft.", StringComparison.InvariantCultureIgnoreCase)
191  && !Assemblies.Any(y => string.Equals(y.FullName, x.FullName, StringComparison.InvariantCultureIgnoreCase))))
