114  private Graph<IMapping> FindGraph(IEnumerable<IMapping> mappings)
115  {
116  var Graph = new Graph<IMapping>();
117  foreach (var Mapping in mappings)
118  {
119  Graph.AddVertex(Mapping);
120  }
121  foreach (var Mapping in mappings)
122  {
123  Type ObjectType = Mapping.ObjectType;
124  Type BaseType = ObjectType.BaseType;
125  while (BaseType != typeof(object)
126  && BaseType != null)
127  {
128  var BaseMapping = mappings.FirstOrDefault(x => x.ObjectType == BaseType);
129  if (BaseMapping != null)
130  {
131  var SinkVertex = Graph.Vertices.First(x => x.Data == BaseMapping);
132  var SourceVertex = Graph.Vertices.First(x => x.Data == Mapping);
133  SourceVertex.AddOutgoingEdge(SinkVertex);
134  }
135  BaseType = BaseType.BaseType;
136  }
137  foreach (Type Interface in ObjectType.GetInterfaces())
138  {
139  var BaseMapping = mappings.FirstOrDefault(x => x.ObjectType == Interface);
140  if (BaseMapping != null)
141  {
142  var SinkVertex = Graph.Vertices.First(x => x.Data == BaseMapping);
143  var SourceVertex = Graph.Vertices.First(x => x.Data == Mapping);
144  SourceVertex.AddOutgoingEdge(SinkVertex);
145  }
146  }
147  }
148  return Graph;
149  }
