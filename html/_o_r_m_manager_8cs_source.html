108  private IEnumerable<IMapping> KahnSort(Graph<IMapping> graph)
109  {
110  var ResultList = new List<IMapping>();
111  var StartingNodes = FindStartingVertices(graph);
112  while (StartingNodes.Count > 0)
113  {
114  var Vertex = StartingNodes.First();
115  StartingNodes.Remove(Vertex);
116  ResultList.AddIfUnique(Vertex.Data);
117  foreach (Edge<IMapping> Edge in Vertex.OutgoingEdges.ToList())
118  {
119  Vertex<IMapping> Sink = Edge.Sink;
120  Edge.Remove();
121  if (Sink.IncomingEdges.Count == 0)
122  {
123  ResultList.AddIfUnique(Sink.Data);
124  StartingNodes.AddIfUnique(Sink);
125  }
126  }
127  }
128  return ResultList;
129  }

148  private void SortMappings()
149  {
150  foreach (IDatabase Database in Mappings.Keys.OrderBy(x => x.Order))
151  {
152  var Order = 0;
153  var Graph = MapperProvider.GetStructure(Database.GetType());
154  foreach (var Mapping in KahnSort(Graph.Copy()).Reverse())
