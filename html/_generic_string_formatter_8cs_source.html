77  public virtual string Format(string Input, string FormatPattern)
78  {

81  var ReturnValue = new StringBuilder();

117  protected virtual string GetMatchingInput(string Input, char FormatChar, out char MatchChar)
118  {
119  bool Digit = FormatChar == DigitChar;
120  bool Alpha = FormatChar == AlphaChar;
121  if (!Digit && !Alpha)
122  {
123  MatchChar = FormatChar;
124  return Input;
125  }
126  int Index = 0;
127  MatchChar = char.MinValue;
128  for (int x = 0; x < Input.Length; ++x)
129  {
130  if ((Digit && char.IsDigit(Input[x])) || (Alpha && char.IsLetter(Input[x])))
131  {
132  MatchChar = Input[x];
133  Index = x + 1;
134  break;
135  }
136  }
137  return Input.Substring(Index);
138  }

145  protected virtual bool IsValid(string FormatPattern)
146  {
147  Contract.Requires<ArgumentNullException>(!string.IsNullOrEmpty(FormatPattern), "FormatPattern");
148  bool EscapeCharFound = false;
149  for (int x = 0; x < FormatPattern.Length; ++x)
150  {
151  if (EscapeCharFound && FormatPattern[x] != DigitChar
152  && FormatPattern[x] != AlphaChar
153  && FormatPattern[x] != EscapeChar)
154  return false;
155  else if (EscapeCharFound)
156  EscapeCharFound = false;
157  else EscapeCharFound |= FormatPattern[x] == EscapeChar;
158  }
159  if (EscapeCharFound)
160  return false;
161  return true;
162  }
