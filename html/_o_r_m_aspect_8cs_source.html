111  public string SetupEndMethod(MethodInfo Method, Type BaseType, string ReturnValueName)
112  {
113  var Builder = new StringBuilder();
114  if (Mapper[BaseType] != null && Method.Name.StartsWith("get_", StringComparison.Ordinal))
115  {
116  foreach (IMapping Mapping in Mapper[BaseType])
117  {
118  var Property = Mapping.Properties.FirstOrDefault(x => x.Name == Method.Name.Replace("get_", ""));
119  if (Property != null)
120  {
121  if (Property is IManyToOne || Property is IMap)
122  Builder.AppendLine(SetupSingleProperty(ReturnValueName, Property));
123  else if (Property is IIEnumerableManyToOne || Property is IManyToMany
124  || Property is IIListManyToMany || Property is IIListManyToOne
125  || Property is ICollectionManyToMany || Property is ICollectionManyToOne)
126  Builder.AppendLine(SetupIEnumerableProperty(ReturnValueName, Property));
127  else if (Property is IListManyToMany || Property is IListManyToOne)
128  Builder.AppendLine(SetupListProperty(ReturnValueName, Property));
129  return Builder.ToString();
130  }
131  }
132  }
133  return Builder.ToString();
134  }
