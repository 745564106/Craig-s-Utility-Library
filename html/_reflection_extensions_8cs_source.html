666  {
667  Contract.Requires<ArgumentNullException>(Property != null, "Property");
668  var PropertyName = Property.PropertyName();
669  var SplitName = PropertyName.Split(new string[] { "." }, StringSplitOptions.RemoveEmptyEntries);
670  if (SplitName.Length == 0)
671  return null;
672  var PropertyInfo = typeof(ClassType).GetProperty(SplitName[0]);
673  var ObjectInstance = Expression.Parameter(PropertyInfo.DeclaringType, "x");
674  var PropertySet = Expression.Parameter(typeof(DataType), "y");
675  var DefaultConstant = Expression.Constant(((object)null).To(PropertyInfo.PropertyType, null), PropertyInfo.PropertyType);
676  MethodCallExpression SetterCall = null;
677  MemberExpression PropertyGet = null;
678  if (SplitName.Length > 1)
679  {
680  PropertyGet = Expression.Property(ObjectInstance, PropertyInfo);
681  for (int x = 1; x < SplitName.Length - 1; ++x)
682  {
683  PropertyInfo = PropertyInfo.PropertyType.GetProperty(SplitName[x]);
684  if (PropertyInfo == null)
685  throw new NullReferenceException("PropertyInfo can't be null");
686  PropertyGet = Expression.Property(PropertyGet, PropertyInfo);
687  }
688  PropertyInfo = PropertyInfo.PropertyType.GetProperty(SplitName[SplitName.Length - 1]);
689  }
690  var SetMethod = PropertyInfo.GetSetMethod();
691  if (SetMethod != null)
692  {
693  if (PropertyInfo.PropertyType != typeof(DataType))
694  {
695  var ConversionMethod = typeof(TypeConversionExtensions).GetMethods().FirstOrDefault(x => x.ContainsGenericParameters
696  && x.GetGenericArguments().Length == 2
697  && x.Name == "To"
698  && x.GetParameters().Length == 2);
699  ConversionMethod = ConversionMethod.MakeGenericMethod(typeof(DataType), PropertyInfo.PropertyType);
700  var Convert = Expression.Call(ConversionMethod, PropertySet, DefaultConstant);
701  SetterCall = PropertyGet == null ? Expression.Call(ObjectInstance, SetMethod, Convert) : Expression.Call(PropertyGet, SetMethod, Convert);
702  return Expression.Lambda<Action<ClassType, DataType>>(SetterCall, ObjectInstance, PropertySet);
703  }
704  SetterCall = PropertyGet == null ? Expression.Call(ObjectInstance, SetMethod, PropertySet) : Expression.Call(PropertyGet, SetMethod, PropertySet);
705  }
706  else
