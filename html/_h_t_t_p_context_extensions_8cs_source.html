46  public static string DumpAllInformation(this HttpContext Context, bool HTMLOutput = false)
47  {
48  if (Context == null)
49  return "";
50  string HTMLTemplate = "<strong>Request Variables</strong><br />{Request}<br /><br /><strong>Response Variables</strong><br />{Response}<br /><br /><strong>Server Variables</strong><br />{Server}<br /><br /><strong>Session Variables</strong><br />{Session}<br /><br /><strong>Cookie Variables</strong><br />{Cookie}<br /><br /><strong>Cache Variables</strong><br />{Cache}<br /><br /><strong>Application Variables</strong><br />{Application}";
51  string NormalTemplate = "Request Variables\r\n{Request}\r\n\r\nResponse Variables\r\n{Response}\r\n\r\nServer Variables\r\n{Server}\r\n\r\nSession Variables\r\n{Session}\r\n\r\nCookie Variables\r\n{Cookie}\r\n\r\nCache Variables\r\n{Cache}\r\n\r\nApplication Variables\r\n{Application}";
52  KeyValuePair<string, string>[] Values = {new KeyValuePair<string,string>("{Request}",Context.Request.DumpRequestVariable(HTMLOutput)),
53  new KeyValuePair<string,string>("{Response}",Context.Response.DumpResponseVariable(HTMLOutput)),
54  new KeyValuePair<string,string>("{Server}",Context.Request.DumpServerVars(HTMLOutput)),
55  new KeyValuePair<string,string>("{Session}",Context.Session.DumpSession(HTMLOutput)),
56  new KeyValuePair<string,string>("{Cookies}",Context.Request.Cookies.DumpCookies(HTMLOutput)),
57  new KeyValuePair<string,string>("{Cache}",Context.Cache.DumpCache(HTMLOutput)),
58  new KeyValuePair<string,string>("{Application}",Context.Application.DumpApplicationState(HTMLOutput))};
59  return HTMLOutput ? HTMLTemplate.ToString(Values) : NormalTemplate.ToString(Values);
60  }

148  public static string DumpCookies(this HttpCookieCollection Input, bool HTMLOutput = false)
149  {
150  Contract.Requires<ArgumentNullException>(Input != null, "Input");
151  var String = new StringBuilder();
152  String.Append(HTMLOutput ? "<table><thead><tr><th>Name</th><th>Sub Name</th><th>Value</th></tr></thead><tbody>" : "Name\t\tSub Name\t\tValue\r\n");
153  foreach (string Key in Input.Keys)
154  {
155  if (Input[Key].Values.Count > 1)
156  {
157  foreach (string SubKey in Input[Key].Values.Keys)
158  {
159  String.Append(HTMLOutput ? "<tr><td>" : "").Append(Key)
160  .Append(HTMLOutput ? "</td><td>" : "\t\t")
161  .Append(SubKey)
162  .Append(HTMLOutput ? "</td><td>" : "\t\t")
163  .Append(Input[Key].Values[SubKey])
164  .Append(HTMLOutput ? "</td></tr>" : "\r\n");
165  }
166  }
167  else
168  {
169  String.Append(HTMLOutput ? "<tr><td>" : "").Append(Key)
170  .Append(HTMLOutput ? "</td><td>" : "\t\t")
171  .Append(HTMLOutput ? "</td><td>" : "\t\t")
172  .Append(Input[Key].Value)
173  .Append(HTMLOutput ? "</td></tr>" : "\r\n");
174  }
175  }
176  String.Append(HTMLOutput ? "</tbody></table>" : "\r\n");
177  return String.ToString();
178  }
