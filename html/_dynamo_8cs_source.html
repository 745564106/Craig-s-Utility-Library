754  protected virtual object GetValue(string Name, Type ReturnType)
755  {
756  var Value = RaiseGetValueStart(Name);
757  if (Value != null)
758  return Value;
759  if (ContainsKey(Name))
760  {
761  if (InternalValues.TryGetValue(Name, out Value))
762  return Value.To(ReturnType, null);
763  }
764  if (!ChildValues.ContainsKey(Name))
765  {
766  var ObjectType = GetType();
767  var Property = ObjectType.GetProperty(Name);
768  if (Property != null)
769  {
770  var Temp = Property.PropertyGetter<Dynamo>().Compile();
771  ChildValues.AddOrUpdate(Name, x => () => Temp(this), (x, y) => () => Temp(this));
772  }
773  else
774  ChildValues.AddOrUpdate(Name, x => () => null, (x, y) => null);
775  }
776  var ReturnValue = ChildValues[Name]().To(ReturnType, null);
777  Value = RaiseGetValueEnd(Name, ReturnValue);
778  return Value ?? ReturnValue;
779  }
