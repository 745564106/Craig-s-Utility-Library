80  public override IBatch CascadeDelete(ClassType Object, ISourceInfo Source, IList<object> ObjectsSeen)
81  {
82  var Provider = IoC.Manager.Bootstrapper.Resolve<QueryProvider.Manager>();
83  var MappingProvider = IoC.Manager.Bootstrapper.Resolve<Mapper.Manager>();
84  IMapping PropertyMapping = MappingProvider[typeof(DataType), Source];
85  var Batch = Provider.Batch(Source);
86  var AspectObject = Object as IORMObject;
87  if (Object == null || (AspectObject != null && ObjectsSeen.Contains(Mapping.IDProperties.FirstOrDefault().GetValue(Object))))
88  return Provider.Batch(Source);
89  if (AspectObject != null)
90  ObjectsSeen.Add(Mapping.IDProperties.FirstOrDefault().GetValue(Object));
91  var List = CompiledExpression(Object);
92  if (List == null)
93  return Batch;
94  foreach (DataType Item in List.Where(x => x != null))
95  {
96  foreach (IProperty<DataType> Property in PropertyMapping.Properties.Where(x => x.Cascade))
97  {
98  Batch.AddCommand(Property.CascadeDelete(Item, Source, ObjectsSeen.ToList()));
99  }
100  Batch.AddCommand(Provider.Generate<DataType>(Source, PropertyMapping, Structure).Delete(Item));
101  }
102  Utilities.IoC.Manager.Bootstrapper.Resolve<DataTypes.Caching.Manager>().Cache().RemoveByTag(typeof(DataType).GetName());
103  return Batch;
104  }
