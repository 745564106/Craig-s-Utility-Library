253  public virtual IDirectory CopyTo(IDirectory Directory, CopyOptions Options = CopyOptions.CopyAlways)
254  {
255  if (InternalDirectory == null || Directory == null)
256  return this;
257  Directory.Create();
258  foreach (IFile TempFile in EnumerateFiles())
259  {
260  switch (Options)
261  {
262  case CopyOptions.CopyAlways:
263  TempFile.CopyTo(Directory, true);
264  break;
265  case CopyOptions.CopyIfNewer:
266  if (new FileInfo(Directory.FullName + "\\" + TempFile.Name.Replace("/", "").Replace("\\", ""), UserName, Password, Domain).Exists)
267  {
268  var FileInfo = new FileInfo(Directory.FullName + "\\" + TempFile.Name.Replace("/", "").Replace("\\", ""), UserName, Password, Domain);
269  if (FileInfo.Modified.CompareTo(TempFile.Modified) < 0)
270  TempFile.CopyTo(Directory, true);
271  }
272  else
273  {
274  TempFile.CopyTo(Directory, true);
275  }
276 
277  break;
278  case CopyOptions.DoNotOverwrite:
279  TempFile.CopyTo(Directory, false);
280  break;
281  }
282  }
283  foreach (IDirectory SubDirectory in EnumerateDirectories())
284  SubDirectory.CopyTo(new DirectoryInfo(Directory.FullName + "\\" + SubDirectory.Name.Replace("/", "").Replace("\\", ""), UserName, Password, Domain), Options);
285  return Directory;
286  }
