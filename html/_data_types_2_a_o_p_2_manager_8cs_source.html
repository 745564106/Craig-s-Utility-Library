116  var AssembliesUsing = new List<Assembly>();

118  Aspects.ForEach(x => AssembliesUsing.AddIfUnique(x.AssembliesUsing));
119

122  Usings.Add("System.Collections.Generic");
123  Usings.Add("System.Linq");
124  Usings.Add("System.Text");
125  Usings.Add("System.Threading.Tasks");

127 
128  var Interfaces = new List<Type>();
129  Aspects.ForEach(x => Interfaces.AddRange(x.InterfacesUsing ?? new List<Type>()));
130 
131  var Builder = new StringBuilder();
132 
133  foreach (Type Type in TempTypes)
134  {
135  AssembliesUsing.AddIfUnique(Type.Assembly);
136  AssembliesUsing.AddIfUnique(GetAssemblies(Type));
137 
138  string Namespace = "CULGeneratedTypes.C" + Guid.NewGuid().ToString("N");
139  string ClassName = Type.Name + "Derived";
140  Builder.AppendLine(Setup(Type, Namespace, ClassName, Usings, Interfaces, AssembliesUsing));
141  }

145  foreach (Type Type in TempTypes)

147  Manager.Classes.AddOrUpdate(Type,

190  private static Assembly[] GetAssemblies(Type Type)
191  {
192  var Types = new List<Assembly>();
193  Type TempType = Type;
194  while (TempType != null)
195  {
196  Types.AddIfUnique(TempType.Assembly.GetReferencedAssemblies().ForEach(x =>
197  {
198  try
199  {
200  return Assembly.Load(x);
201  }
202  catch
203  {
204  return null;
205  }
206  }).Where(x => x != null));
207  Types.AddIfUnique(TempType.Assembly);
208  TempType.GetInterfaces().ForEach(x => Types.AddIfUnique(GetAssembliesSimple(x)));
209  TempType.GetEvents().ForEach(x => Types.AddIfUnique(GetAssembliesSimple(x.EventHandlerType)));
210  TempType.GetFields().ForEach(x => Types.AddIfUnique(GetAssembliesSimple(x.FieldType)));
211  TempType.GetProperties().ForEach(x => Types.AddIfUnique(GetAssembliesSimple(x.PropertyType)));
212  TempType.GetMethods().ForEach(x =>
213  {
214  Types.AddIfUnique(GetAssembliesSimple(x.ReturnType));
215  x.GetParameters().ForEach(y => Types.AddIfUnique(GetAssembliesSimple(y.ParameterType)));
216  });
217  TempType = TempType.BaseType;
218  if (TempType == typeof(object))
219  break;
220  }
221  return Types.ToArray();
222  }

234  break;
235  }

272  public class {2} : {3}{4} {5}

274 ", usings.ToString(x => "using " + x + ";", "\r\n"),
275  @namespace,

278  interfaces.Count > 0 ? "," : "", interfaces.ToString(x => x.Name));

281  //Builder.AppendLine(new ConstructorGenerator(type.GetConstructors(BindingFlags.Public | BindingFlags.Instance).First(x => x.GetParameters().Length == 0)).Generate(assembliesUsing, Aspects));
282  }
283 
284  Aspects.ForEach(x => Builder.AppendLine(x.SetupInterfaces(type)));
285 
286  Type TempType = type;

288  while (TempType != null)

304  MethodsAlreadyDone.Add(GetMethodInfo.Name);

306  }
307  else if (!MethodsAlreadyDone.Contains("get_" + Property.Name)
308  && GetMethodInfo != null
309  && GetMethodInfo.IsVirtual
310  && SetMethodInfo == null
311  && !GetMethodInfo.IsFinal
312  && Property.GetIndexParameters().Length == 0)

314  Builder.AppendLine(new PropertyGenerator(Property).Generate(assembliesUsing, Aspects));
315  MethodsAlreadyDone.Add(GetMethodInfo.Name);

320  MethodsAlreadyDone.Add(GetMethodInfo.Name);

322  MethodsAlreadyDone.Add(SetMethodInfo.Name);

324  }

351  private static string SetupMethod(Type Type, MethodInfo MethodInfo, bool IsProperty)
352  {
353  if (MethodInfo == null)
354  return "";
355  var Builder = new StringBuilder();
356  var BaseMethodName = MethodInfo.Name.Replace("get_", "").Replace("set_", "");
357  string ReturnValue = MethodInfo.ReturnType != typeof(void) ? "FinalReturnValue" : "";
358  string BaseCall = "";
359  if (IsProperty)
360  BaseCall = string.IsNullOrEmpty(ReturnValue) ? "base." + BaseMethodName : ReturnValue + "=base." + BaseMethodName;
361  else
362  BaseCall = string.IsNullOrEmpty(ReturnValue) ? "base." + BaseMethodName + "(" : ReturnValue + "=base." + BaseMethodName + "(";
363  var Parameters = MethodInfo.GetParameters();
364  if (IsProperty)
365  {
366  BaseCall += Parameters.Length > 0 ? "=" + Parameters.ToString(x => x.Name) + ";" : ";";
367  }
368  else
369  {
370  BaseCall += Parameters.Length > 0 ? Parameters.ToString(x => (x.IsOut ? "out " : "") + x.Name) : "";
371  BaseCall += ");\r\n";
372  }
373  Builder.AppendLineFormat(@"
374  try
375  {{
376  {0}
377  {1}
378  {2}
379  {3}
380  {4}
381  }}
382  catch(Exception CaughtException)
383  {{
384  {5}
385  throw;
386  }}",
387  MethodInfo.ReturnType != typeof(void) ? MethodInfo.ReturnType.GetName() + " " + ReturnValue + ";" : "",
388  Aspects.ForEach(x => x.SetupStartMethod(MethodInfo, Type)).ToString(x => x, "\r\n"),
389  BaseCall,
390  Aspects.ForEach(x => x.SetupEndMethod(MethodInfo, Type, ReturnValue)).ToString(x => x, "\r\n"),
391  string.IsNullOrEmpty(ReturnValue) ? "" : "return " + ReturnValue + ";",
392  Aspects.ForEach(x => x.SetupExceptionMethod(MethodInfo, Type)).ToString(x => x, "\r\n"));
393  return Builder.ToString();
394  }
