78  private static void SetupViews(ITable table, dynamic item)
79  {
80  if (item == null)
81  throw new ArgumentNullException(nameof(item));
82  if (table == null)
83  throw new ArgumentNullException(nameof(table));
84  var View = (View)table;
85  View.Definition = item.Definition;
86  string ColumnName = item.Column;
87  string ColumnType = item.COLUMN_TYPE;
88  int MaxLength = item.MAX_LENGTH;
89  if (ColumnType == "nvarchar")
90  MaxLength /= 2;
91  bool Nullable = item.IS_NULLABLE;
92  View.AddColumn<string>(ColumnName, ColumnType.To<string, SqlDbType>().To(DbType.Int32), MaxLength, Nullable);
