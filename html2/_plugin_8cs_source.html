212  private void CopyTo(DirectoryInfo Directory1, DirectoryInfo Directory2)
213  {

216  Directory2.Create();

218  {
219  try
220  {
221  TempFile.CopyTo(Path.Combine(Directory2.FullName, TempFile.Name), true);
222  }
223  catch { }
224  }
225  foreach (DirectoryInfo SubDirectory in Directory1.EnumerateDirectories())
226  CopyTo(SubDirectory, new DirectoryInfo(Path.Combine(Directory2.FullName, SubDirectory.Name)));
